- 判断一个 9x9 的数独是否有效。只需要根据以下规则，验证已经填入的数字是否有效即可。

  1. 数字 1-9 在每一行只能出现一次。
  2. 数字 1-9 在每一列只能出现一次。
  3. 数字 1-9 在每一个以粗实线分隔的 3x3 宫内只能出现一次。
  
  - 数独部分空格内已填入了数字，空白格用 '.' 表示。

    **示例 1:**
    ```
    输入: 
    [
      ["5","3",".",".","7",".",".",".","."],
      ["6",".",".","1","9","5",".",".","."],
      [".","9","8",".",".",".",".","6","."],
      ["8",".",".",".","6",".",".",".","3"],
      ["4",".",".","8",".","3",".",".","1"],
      ["7",".",".",".","2",".",".",".","6"],
      [".","6",".",".",".",".","2","8","."],
      [".",".",".","4","1","9",".",".","5"],
      [".",".",".",".","8",".",".","7","9"]
    ]
    输出: true
    ```
    **示例 2:**
    
    ```
    输入: 
    [
      ["8","3",".",".","7",".",".",".","."],
      ["6",".",".","1","9","5",".",".","."],
      [".","9","8",".",".",".",".","6","."],
      ["8",".",".",".","6",".",".",".","3"],
      ["4",".",".","8",".","3",".",".","1"],
      ["7",".",".",".","2",".",".",".","6"],
      [".","6",".",".",".",".","2","8","."],
      [".",".",".","4","1","9",".",".","5"],
      [".",".",".",".","8",".",".","7","9"]
    ]
    输出: false
    ```

    **说明:**

    - 一个有效的数独（部分已被填充）不一定是可解的。
    - 只需要根据以上规则，验证已经填入的数字是否有效即可。
    - 给定数独序列只包含数字 1-9 和字符 '.' 。
    - 给定数独永远是 9x9 形式的。

///这是最早我敲的，时间复杂度太高了，超时不能通过
class Solution {
public:
    //判断竖直方向是否有效
    bool isVertical(vector<vector<char>>& board, int i, int j) {
        for (int k = 0; k < 9; k++) {
            if (board[i][j] == board[k][j]) {
                if (k == i) {
                    continue;
                }
                return false;
            }
        }
        return true;
    }
    //判断水平方向是否有效
    bool isHorizontal(vector<vector<char>>& board, int i, int j) {
        for (int k = 0; k < 9; k++) {
            if (board[i][j] == board[i][k]) {
                if (k == j) {
                    continue;
                }
                return false;
            }
        }
        return true;
    }
    //判断九宫格内是否有效
    bool isSudoku(vector<vector<char>>& board, int i, int j) {
        int x = (i / 3) * 3, y = (j / 3) * 3;
        for (int k = x; k < x + 3; k++) {
            for (int g = y; g < y + 3; y++) {
                if (k == i && g == j) {
                    continue;
                }
                if (board[k][g] == board[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }
    bool isValidSudoku(vector<vector<char>>& board) {
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                if(board[i][j] != '.') {
                    if (!isVertical(board, i, j)) return false;
                    if(!isHorizontal(board, i, j)) return false;
                    if(!isSudoku(board, i, j)) return false;
                }
            }
        }
        return true;
    }
};
